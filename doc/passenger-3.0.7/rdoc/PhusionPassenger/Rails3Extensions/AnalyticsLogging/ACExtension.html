def process_action(action, *args)
        log = request.env[PASSENGER_ANALYTICS_WEB_LOG]
        if log
                log.message("Controller action: #{self.class.name}##{action_name}")
                log.measure("framework request processing") do
                        super
                end
        else
                super
        end
end
